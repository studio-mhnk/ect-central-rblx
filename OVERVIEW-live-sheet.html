  <title>Live Google Sheet Table</title>

  <style>

    table {

      border-collapse: collapse;

      width: 100%;

      font-family: Arial, sans-serif;

    }

    th, td {

      border: 1px solid #ccc;

      padding: 8px;

    }

    th {

      background-color: #f2f2f2;

      font-weight: bold;

    }

    tr:nth-child(even) {background-color: #fafafa;}

    tr:hover {background-color: #f1f1f1;}

  </style>

<body>


<div id="sheetTable">Loading data...</div>


<script>

const sheetId = '1wG1t-qRKPqgBWRePg-WYB7bkjzV70ercdsUtrkzmBXg'; // Your Sheet ID

const sheetName = ‘OVERVIEW’; // Name of the tab you want to show


async function fetchSheetData() {

  try {

    const url = `https://docs.google.com/spreadsheets/d/${sheetId}/gviz/tq?tqx=out:json&sheet=${encodeURIComponent(sheetName)}`;

    const res = await fetch(url);

    const text = await res.text();

    const json = JSON.parse(text.substr(47).slice(0, -2)); 


    const cols = json.table.cols.map(c => c.label || "");

    const rows = json.table.rows;


    let html = '<table><thead><tr>' + cols.map(c => `<th>${c}</th>`).join('') + '</tr></thead><tbody>';

    rows.forEach(r => {

      html += '<tr>';

      r.c.forEach(cell => html += `<td>${cell && cell.v !== null ? cell.v : ''}</td>`);

      html += '</tr>';

    });

    html += '</tbody></table>';


    document.getElementById('sheetTable').innerHTML = html;

  } catch (e) {

    document.getElementById('sheetTable').innerHTML = 'Error loading data';

    console.error(e);

  }

}


fetchSheetData();

setInterval(fetchSheetData, 5000); // Refresh every 5 seconds

</script>


</body>

</html>
